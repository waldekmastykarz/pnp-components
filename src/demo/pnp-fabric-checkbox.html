  document.getElementById('pnp-fabric-checkbox').addEventListener('onClick', function (event) {
    alert('Is checked: ' + event.detail);
  });

  document.getElementById('btnCheck').addEventListener('click', function () {
    var checkbox = document.getElementById('pnp-fabric-checkbox');
    checkbox.checked = !checkbox.checked;
  });
</script>
